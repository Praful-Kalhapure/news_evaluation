        <script>
          function previewURL() {
            var url = document.getElementById("url").value;
            var previewFrame = document.getElementById("previewFrame");
            var errorContainer = document.getElementById("errorContainer");
    
            // Reset previous state
            document.getElementById("previewContainer").style.display = "none";
            errorContainer.style.display = "none";
    
            // Validate the URL
            if (!isValidURL(url)) {
              errorContainer.innerHTML = "Invalid URL. Please enter a valid URL.";
              errorContainer.style.display = "block";
              return;
            }
    
            // Set the source of the iframe to the entered URL
            previewFrame.src = url;
    
            // Display the preview container and the analyze button
            document.getElementById("previewContainer").style.display = "block";
            document.getElementById("analyzeButton").style.display = "block";
          }
    
          function clearInput() {
            document.getElementById("url").value = "";
          }
    
          function analyzeContent() {
            document.querySelector('form').submit();
          }
    
    
    
          function isValidURL(url) {
            try {
              new URL(url);
              return true;
            } catch (error) {
              return false;
            }
          }
